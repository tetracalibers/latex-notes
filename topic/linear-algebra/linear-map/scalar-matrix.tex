\begin{definition*}{行列のスカラー倍}
  $A$を行列、$c$をスカラーとするとき、$A$のすべての成分を$c$倍して得られる行列を$cA$とする
\end{definition*}

\begin{theorem*}{行列の積とスカラー倍の性質}
  行列$A,\,B$の積$AB$が定義できるとき、つまり$A$の列の個数と$B$の行の個数が同じであるとき、$c \in \mathbb{R}$に対して
  \begin{equation*}
    (cA)B = A(cB) = c(AB)
  \end{equation*}
  が成り立つ
\end{theorem*}
