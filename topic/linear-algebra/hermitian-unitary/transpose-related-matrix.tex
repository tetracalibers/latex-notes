\section{対称行列と交代行列}
\marginnote{\refbookF p30}

転置に関するさまざまな性質をもつ行列には、それぞれ名前が付けられている

\begin{definition}{対称行列}
  正方行列$A$が次を満たすとき、$A$を\keyword{対称行列}という
  \begin{equation*}
    \transpose{A} = A
  \end{equation*}
\end{definition}

\begin{definition}{交代行列}
  正方行列$A$が次を満たすとき、$A$を\keyword{交代行列}という
  \begin{equation*}
    \transpose{A} = -A
  \end{equation*}
\end{definition}
